```bash
#include <stdio.h>

int main() {
    int i; // declaraction
    printf("%d\n",i);
    
    i=5; // initialisation
    printf("%d\n",i);
    
    i=1.2; // 1.2 will be changed to 1
    printf("%d\n",i);
    
    char a; // single character  
    printf("%c\n",a);
    
    a ='a';
    printf("%c\n",a);
    
    char s[]="abcd"; // Group of characters (string)
    printf("%s\n",s);
    
    int ascii='A'; // get the ascii code
    printf("%d\n",ascii);
    
    float y = 12.5; // Float type
    printf("%f\n",y);
    
    double z = 12.345678; // more precise than float
    printf("%lf\n",z);
}
```
